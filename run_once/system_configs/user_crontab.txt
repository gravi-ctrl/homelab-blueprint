SHELL=/bin/bash
BASH_ENV=/opt/scripts/.env

# .------------ minute (0 - 59)
# | .---------- hour (0 - 23)
# | | .-------- day of month (1 - 31)
# | | | .------ month (1 - 12)
# | | | | .---- day of week (0 - 7) (Sunday=0 or 7)
# | | | | |
# * * * * *
# Notify on Server Boot
@reboot sleep 60 && cron-guard --mode all "Server Startup" "echo 'Host OS has booted. Docker services should be initializing.'"

# Update NextDNS IP and Healthchecks.io server/internet monitor
*/5 * * * * curl -Z -fsS --retry 3 "$SERVER_HC_URL" "$NEXTDNS_URL"

# Sync Personal and Work Obsidian vaults to Git
*/30 * * * * cron-guard --mode fail "Obsidian Notes Sync" "$SCRIPTS_DIR/git-auto-sync.py '$DATA_DIR/syncthing/Backup/obsidian-notes/personal' 'Obsidian Personal'; $SCRIPTS_DIR/git-auto-sync.py '$DATA_DIR/syncthing/Backup/obsidian-notes/work' 'Obsidian Work'"

# Disk Space Alerts
0 * * * * cron-guard --mode fail "Disk Monitor" "$SCRIPTS_DIR/disk-monitor.sh"

# Rotate backups (retain recent) and purge download watch folder
0 0,12 * * * cron-guard --mode fail "Cleanup Job" "$SCRIPTS_DIR/cleanup_script.py $DATA_DIR/syncthing/Backup/self-hosted/contacts-calendars_backup $DATA_DIR/syncthing/Backup/antennapod $DATA_DIR/downloads/watch::DELETE_ALL"

# Update Pi-hole Gravity (adlists) and apply mmotti regex rules
0 1 * * * cron-guard --mode fail "Pi-hole Gravity & Regex" "docker exec pihole /bin/bash -c 'curl -sSL https://raw.githubusercontent.com/mmotti/pihole-regex/master/install.py | python3 && pihole -g'"

# Export Nextcloud Calendars (.ics) and Contacts (.vcf)
0 4 * * * cron-guard --mode fail "Nextcloud Cal/Card Backup" "docker exec calcardbackup /opt/calcardbackup/calcardbackup /var/www/html -o /backup/syncthing/Backup/self-hosted/contacts-calendars_backup -s"

# Apply file renaming rules to Paperless-ngx documents
15 4 * * * cron-guard --mode fail "Paperless Auto Renamer" "docker exec -i paperless-ngx python3 manage.py document_renamer"

# Snapshot system configs/dotfiles and sync `~/scripts`, `~/ctrl_s_master` & `/opt/stacks` to Git
0 5 * * * cron-guard --mode fail "Scripts & System Configs Backup" "$SCRIPTS_DIR/backup-scripts-git.sh"

# Scan docker-compose files and open dependency update PRs on Codeberg
0 9 * * * cron-guard --mode fail "Renovate" "docker run --rm --env-file /opt/stacks/renovate/.env -v /opt/stacks/renovate/config.js:/usr/src/app/config.js renovate/renovate"

# Runs a Borgmatic backup
0 2 * * 0 cron-guard --mode all "Borg Backup" "borgmatic"

# cert-manager: regenerate & upload SSL certs to NPM
0 3 1 * * cron-guard --mode fail "cert-mgr: Renew SSL certs" "$SCRIPTS_DIR/cert-manager/cert-manager.sh renew"
