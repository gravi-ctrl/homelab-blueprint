SHELL=/bin/bash
BASH_ENV=/opt/scripts/.env

# .------------ minute (0 - 59)
# | .---------- hour (0 - 23)
# | | .-------- day of month (1 - 31)
# | | | .------ month (1 - 12)
# | | | | .---- day of week (0 - 7) (Sunday=0 or 7)
# | | | | |
# * * * * *
# NextDNS IP and Healthchecks.io monitor
*/5 * * * * curl -Z -fsS --retry 3 "$SERVER_HC_URL" "$NEXTDNS_URL"

# Pulls and deploys stack updates done by Renovate
*/15 * * * * cron-guard --mode fail "GitOps Deployer" "/opt/scripts/gitops-deploy.sh"

# Sync Personal and Work Obsidian vaults to Git
*/30 * * * * cron-guard --mode fail "Obsidian Notes Sync" "/opt/scripts/git-auto-sync.py '$DATA_DIR/syncthing/Backup/obsidian-notes/personal' 'Obsidian Personal'; /opt/scripts/git-auto-sync.py '$DATA_DIR/syncthing/Backup/obsidian-notes/work' 'Obsidian Work'"

# Disk Space Alerts
0 * * * * cron-guard --mode fail "Disk Monitor" "/opt/scripts/disk-monitor.sh"

# Rotate backups and purge download watch folder
0 0,12 * * * cron-guard --mode fail "Cleanup Job" "/opt/scripts/cleanup_script.py $DATA_DIR/syncthing/Backup/self-hosted/contacts-calendars_backup $DATA_DIR/syncthing/Backup/antennapod $DATA_DIR/downloads/watch::DELETE_ALL"

# Update Pi-hole Gravity and apply mmotti regex rules
55 3 * * * cron-guard --mode fail "Pi-hole Gravity & Regex" "docker exec pihole /bin/bash -c 'curl -sSL https://raw.githubusercontent.com/mmotti/pihole-regex/master/install.py | python3 && pihole -g'"

# Export Nextcloud Calendars and Contacts
10 4 * * * cron-guard --mode fail "Nextcloud CalCard Backup" "docker exec calcardbackup /opt/calcardbackup/calcardbackup /var/www/html -o /backup/syncthing/Backup/self-hosted/contacts-calendars_backup -s"

# Apply file renaming rules to Paperless-ngx documents
25 4 * * * cron-guard --mode fail "Paperless Auto Renamer" "docker exec -i paperless-ngx python3 manage.py document_renamer"

# Snapshot system state, build dashboards, and sync repos to Git
0 5 * * * cron-guard --mode fail "Scripts & System Configs Backup" "/opt/scripts/backup-scripts-git.sh"

# Scan docker-compose files and open dependency update PRs on Codeberg
0 9 * * * cron-guard --mode fail "Renovate" "docker run --rm --env-file $STACKS_DIR/renovate/.env -v $STACKS_DIR/renovate/config.js:/usr/src/app/config.js renovate/renovate"

# Runs a Borgmatic backup
0 2 * * 0 cron-guard --mode all "Borg Backup" "borgmatic"

# cert-manager: regenerate & upload SSL certs to NPM
0 3 1 * * cron-guard --mode fail "cert-mgr: Renew SSL certs" "/opt/scripts/cert-manager/cert-manager.sh renew"
