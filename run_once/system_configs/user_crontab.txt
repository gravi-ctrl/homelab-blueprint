SHELL=/bin/bash
BASH_ENV=/opt/ctrl/.env

# .------------ minute (0 - 59)
# | .---------- hour (0 - 23)
# | | .-------- day of month (1 - 31)
# | | | .------ month (1 - 12)
# | | | | .---- day of week (0 - 7) (Sunday=0 or 7)

# Healthchecks.io Server/Internet monitor
*/5 * * * * cron-guard --mode mute "HC.io Ping" "curl -fsS --retry 3 '$SERVER_HC_URL'"

# Keeps NextDNS filtering pinned to our current public IP
*/5 * * * * cron-guard --mode fail "NextDNS IP Update" "curl -fsS --retry 3 '$NEXTDNS_URL'"

# Sync Personal and Work Obsidian vaults to Git
*/30 * * * * cron-guard --mode fail "Obsidian Notes Sync" "/opt/ctrl/git-auto-sync.py '$DATA_DIR/syncthing/Backup/obsidian-notes/personal' 'Obsidian Personal'; /opt/ctrl/git-auto-sync.py '$DATA_DIR/syncthing/Backup/obsidian-notes/work' 'Obsidian Work'"

# Check if root (/) or /data exceeds 90%
0 * * * * cron-guard --mode fail "Disk Monitor" "/opt/ctrl/disk-monitor.sh"

# Rotate backups (retain recent) and purge download watch folder
0 0,12 * * * cron-guard --mode fail "Cleanup Job" "/opt/ctrl/cleanup_script.py $DATA_DIR/syncthing/Backup/self-hosted/contacts-calendars_backup $DATA_DIR/syncthing/Backup/antennapod $DATA_DIR/downloads/watch::DELETE_ALL"

# Pulls and deploys stack updates done by Renovate
0 3 * * * cron-guard --mode fail "GitOps Deployer" "/opt/ctrl/gitops-deploy.sh"

# Update Pi-hole Gravity and apply mmotti regex rules
0 4 * * * cron-guard --mode fail "Pi-hole Gravity & Regex" "docker exec pihole /bin/bash -c 'curl -sSL https://raw.githubusercontent.com/mmotti/pihole-regex/master/install.py | python3 && pihole -g'"

# Export Nextcloud Calendars (.ics) and Contacts (.vcf)
15 4 * * * cron-guard --mode fail "Nextcloud CalCard Backup" "docker exec calcardbackup /opt/calcardbackup/calcardbackup /var/www/html -o /backup/syncthing/Backup/self-hosted/contacts-calendars_backup -s"

# Apply file renaming rules to Paperless-ngx documents
30 4 * * * cron-guard --mode fail "Paperless Auto Renamer" "docker exec -i paperless-ngx python3 manage.py document_renamer"

# Snapshot system state, build UI dashboards, and sync repos to Git
0 5 * * * cron-guard --mode fail "Scripts & System Configs Backup" "/opt/ctrl/backup-scripts-git.sh"

# Scan docker-compose files and open dependency update PRs on Codeberg
0 9 * * * cron-guard --mode fail "Renovate" "docker run --rm --env-file $STACKS_DIR/renovate/.env -v $STACKS_DIR/renovate/config.js:/usr/src/app/config.js renovate/renovate"

# Runs a Borgmatic backup
0 2 * * 0 cron-guard --mode all "Borg Backup" "borgmatic"

# cert-manager: regenerate & upload SSL certs to NPM
0 3 1 * * cron-guard --mode fail "cert-mgr: Renew SSL certs" "/opt/ctrl/cert-manager/cert-manager.sh renew"
